## 📄 File: `CookieServlet.java`

📍 `src/com/example/`

```java
package com.example;

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class CookieServlet extends HttpServlet {

    protected void doPost(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {

        String lang = req.getParameter("lang");

        Cookie c = new Cookie("language", lang);
        res.addCookie(c);

        PrintWriter out = res.getWriter();
        out.println("<html><body>");
        out.println("Cookie Added: " + lang);
        out.println("<br><a href='cookie'>View Recommendations</a>");
        out.println("</body></html>");
    }

    protected void doGet(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {

        Cookie[] cookies = req.getCookies();

        PrintWriter out = res.getWriter();
        out.println("<html><body>");

        if (cookies != null) {
            for (Cookie c : cookies) {
                if (c.getName().equals("language")) {
                    out.println("Recommended Book for " + c.getValue() + "<br>");
                }
            }
        }

        out.println("</body></html>");
    }
}
```

## 📄 File: `FAQServlet.java`

📍 `src/com/example/`

```java
package com.example;

import java.io.*;
import java.sql.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class FAQServlet extends HttpServlet {

    protected void doGet(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {

        PrintWriter out = res.getWriter();

        try {
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(
                "jdbc:mysql://localhost:3306/faqdb", "root", "");

            Statement st = con.createStatement();

            ResultSet rs = st.executeQuery(
                "SELECT Topics.topicName, FAQ.question, FAQ.answer " +
                "FROM Topics JOIN FAQ ON Topics.topicID = FAQ.topicID");

            out.println("<html><body>");

            while (rs.next()) {
                out.println("<h3>" + rs.getString(1) + "</h3>");
                out.println("Q: " + rs.getString(2) + "<br>");
                out.println("A: " + rs.getString(3) + "<br><br>");
            }

            out.println("</body></html>");
            con.close();

        } catch (Exception e) {
            out.println(e);
        }
    }
}
```

package com.example;

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class SessionServlet extends HttpServlet {

    protected void doPost(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {

        HttpSession session = req.getSession();

        String lang = req.getParameter("lang");

        session.setAttribute("language", lang);

        PrintWriter out = res.getWriter();

        out.println("<html><body>");
        out.println("Session ID: " + session.getId() + "<br>");
        out.println("Language: " + lang + "<br>");
        out.println("Creation Time: " + session.getCreationTime() + "<br>");
        out.println("Last Accessed: " + session.getLastAccessedTime());
        out.println("</body></html>");
    }
}

## 📄 File: `SurveyServlet.java`

```java
package com.example;

import java.io.*;
import java.sql.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class SurveyServlet extends HttpServlet {

    protected void doPost(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {

        String animal = req.getParameter("animal");

        PrintWriter out = res.getWriter();

        try {
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(
                "jdbc:mysql://localhost:3306/surveydb", "root", "");

            Statement st = con.createStatement();

            st.executeUpdate("UPDATE survey SET votes = votes + 1 WHERE animal='" + animal + "'");

            ResultSet rs = st.executeQuery("SELECT * FROM survey");

            int total = 0;
            while (rs.next()) total += rs.getInt("votes");

            rs = st.executeQuery("SELECT * FROM survey");

            out.println("<html><body>");

            while (rs.next()) {
                int votes = rs.getInt("votes");
                double percent = (votes * 100.0) / total;
                out.println(rs.getString("animal") + " : " + percent + "%<br>");
            }

            out.println("Total Votes: " + total);
            out.println("</body></html>");

        } catch (Exception e) {
            out.println(e);
        }
    }
}
```
