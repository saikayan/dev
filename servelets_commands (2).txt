CREATE TABLE topics (
topicID INT PRIMARY KEY AUTO_INCREMENT,
topicName VARCHAR(100)
);



CREATE TABLE faq (
id INT PRIMARY KEY AUTO_INCREMENT,
topicID INT,
question VARCHAR(255),
answer VARCHAR(255),
FOREIGN KEY (topicID) REFERENCES topics(topicID)
);
